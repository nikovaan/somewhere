#include <iostream>
#include <fstream>

using namespace std;

void tulosta_matriisi(int matriisi[5][5]);
int laske_summa(int matriisi[5][5]);

int main(void)
{
  int matriisi[5][5];
  int summa;
  ifstream luku("matriisi.txt");
  if (!luku) {
    cout << "Tiedoston avaaminen epÃ¤onnistui...";
  }
  else {
    for (int y=0; y<5;y++){
      for (int x=0;x<5;x++){
        luku >> matriisi[y][x];
      }
    }
    luku.close();
    cout << "Matriisi:" << endl;
    tulosta_matriisi(matriisi);
    summa = laske_summa(matriisi);
    cout << "Alkioiden summa: " << summa << endl;
  }
}

void tulosta_matriisi(int matriisi[5][5])
{
	int i1, i2;

	for (i1 = 0; i1 < 5; i1++)
	{
		for (i2 = 0; i2 < 5; i2++)
		{
			cout << matriisi[i1][i2] << " ";
		}
		cout << endl;
	}
}

int laske_summa(int matriisi[5][5])
{
	int _summa = 0;
	int i1, i2;

	for (i1 = 0; i1 < 5; i1++)
	{
		for (i2 = 0; i2 < 5; i2++)
		{
			_summa = _summa + matriisi[i1][i2];
		}
	}

	return _summa;
}
